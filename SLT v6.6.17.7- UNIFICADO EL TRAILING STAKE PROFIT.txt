// - Requiere N = cmBarsLookback velas con dirección consistente
// - Umbral opcional: cmChangeThreshold (valor o %), activado si > 0
// Devuelve: true si cumple dirección (N velas) y magnitud (si aplica)

changeMomentumPro(dir) =>
    changeOK = true
    for i = 0 to cmBarsLookback - 1
        if dir == "up"
            changeOK := changeOK and val[i] > val[i+1]
        else
            changeOK := changeOK and val[i] < val[i+1]
    if cmChangeThreshold > 0
        diff  = math.abs(val - val[cmBarsLookback])
        limit = cmUsePercentChange ? math.abs(val[cmBarsLookback]) * ((cmChangeThreshold * 0.0001) / 100) : (cmChangeThreshold * 0.0001)
        changeOK := changeOK and diff >= limit
    changeOK

        exitShort := ta.crossover( val, 0) and barsSinceEntryShort >= minBarsToHold
    'Half Momentum' =>
        exitLong  := val >= momentumMin * Umbral_Mom and barsSinceEntryLong  >= minBarsToHold
        exitShort := val <= momentumMax * Umbral_Mom and barsSinceEntryShort >= minBarsToHold
    'Change Momentum' =>
        exitLong  := (val[1] > 0 and val < val[1]) and barsSinceEntryLong  >= minBarsToHold
        exitShort := (val[1] < 0 and val > val[1]) and barsSinceEntryShort >= minBarsToHold
    'Max Momentum' =>
        exitLong  := val >= momentumMax and barsSinceEntryLong  >= minBarsToHold
        exitShort := val <= momentumMin and barsSinceEntryShort >= minBarsToHold
    'Min Momentum' =>
        exitLong  := val <= momentumMin and barsSinceEntryLong  >= minBarsToHold
        exitShort := val >= momentumMax and barsSinceEntryShort >= minBarsToHold
    'Zero Inverse' =>
        exitLong  := ta.crossover( val, 0) and barsSinceEntryLong  >= minBarsToHold
        exitShort := ta.crossunder(val, 0) and barsSinceEntryShort >= minBarsToHold
    'Adaptativa' =>
        exitLong  := (val >= momentumMin * Umbral_Mom or barsSinceEntryLong  > Tiempo_Trans) and barsSinceEntryLong  >= minBarsToHold
        exitShort := (val <= momentumMax * Umbral_Mom or barsSinceEntryShort > Tiempo_Trans) and barsSinceEntryShort >= minBarsToHold

    progressATR_long := (highSinceEntry - strategy.opentrades.entry_price(0)) / atr1

if dirShortOK and strategy.position_size < 0 and not na(lowSinceEntry) and atr1 > 0
    progressATR_short := (strategy.opentrades.entry_price(0) - lowSinceEntry) / atr1

// Distancia objetivo del TTP en ATR (según modo)
ttpDistATR(dir) =>
    distSimple = ttpSimpleDistATR

    reached1 = (dir == "long" ? nz(progressATR_long, 0) : nz(progressATR_short, 0)) >= ttpStep1_triggerATR
    reached2 = (dir == "long" ? nz(progressATR_long, 0) : nz(progressATR_short, 0)) >= ttpStep2_triggerATR
    reached3 = (dir == "long" ? nz(progressATR_long, 0) : nz(progressATR_short, 0)) >= ttpStep3_triggerATR

    distStep =
         ttpMode == "Escalonado 3 pasos" and reached3 ? ttpStep3_distATR :
         ttpMode == "Escalonado 3 pasos" and reached2 ? ttpStep2_distATR :
         ttpMode == "Escalonado 3 pasos" and reached1 ? ttpStep1_distATR :
         ttpMode == "Escalonado 2 pasos" and reached2 ? ttpStep2_distATR :
         ttpMode == "Escalonado 2 pasos" and reached1 ? ttpStep1_distATR :
         distSimple

    ttpMode == "OFF" ? na : (ttpMode == "Simple continuo" ? distSimple : distStep)

// Distancia efectiva (en precio)
float _ttpDistATR_long  = ttpDistATR("long")
float _ttpDistATR_short = ttpDistATR("short")

float ttpDistLong  = na(_ttpDistATR_long)  ? na : atr1 * _ttpDistATR_long
float ttpDistShort = na(_ttpDistATR_short) ? na : atr1 * _ttpDistATR_short

// Gate de activación
bool longTTPReady  = dirLongOK  and ttpMode != "OFF" and not na(progressATR_long)  and progressATR_long  >= ttpActivateATR
bool shortTTPReady = dirShortOK and ttpMode != "OFF" and not na(progressATR_short) and progressATR_short >= ttpActivateATR

// Reglas de salida por TTP (pullback desde el extremo favorable)
if dirLongOK and strategy.position_size > 0 and longTTPReady and not na(ttpDistLong) and not na(highSinceEntry)
    if close <= highSinceEntry - ttpDistLong
        // --- BYBIT CLOSE 100%
        f_bybit_close_100()
        strategy.close("Long", comment="Trailing Take Profit Long")
